
add_namespace = spain

#### Spanish Civil War ####

country_event = {
	id = spain.1
	title = "Spanish war"
	desc = "A second civil war comme in Spain. We are now alone and we can't count on European powers which are under dictatorship."
	picture = GFX_report_event_generic_read_write
	
	fire_only_once = yes
	
	trigger = { 
		tag = SPR
		date > 1936.2.2
	}

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		SPR_setup_civil_war = yes

		every_owned_state = {
			add_core_of = SPC
			add_claim_by = SPC
			teleport_armies = {
				to_state_array = owned_controlled_states
			}
		}

		every_owned_state = {
			add_core_of = SPA
			add_claim_by = SPA
			teleport_armies = {
				to_state_array = owned_controlled_states
			}
		}

		SPR = {
			transfer_technology = yes
		}
	}

	option = {
		name = "We shall fight them!"
		
	        declare_war_on = {
		                target = SPC
		                type = annex_everything
	        }
	        declare_war_on = {
		                target = SPA
		                type = annex_everything
	        }
	        SPA = { declare_war_on = {
		                target = SPC
		                type = annex_everything
	        } }
		SPR = {
			every_owned_state = {
				add_claim_by = SPC
			}
		}
		SPC = {
			every_owned_state = {
				add_claim_by = SPA
			}
		}
		SPR = {
			every_owned_state = {
				add_claim_by = SPA
			}
		}

		SPA = { set_politics = { ruling_party = fascism } }
		SPC = { set_politics = { ruling_party = anarchist } }
	}
}
