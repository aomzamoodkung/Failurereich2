
add_namespace = poland

country_event = {
	id = poland.1
	title = "Hungarish civil war"
	desc = "A coup was organized in our regions, especially Slovakia, where upprising come."
	picture = GFX_report_event_greater_hungary2
	
	is_triggered_only = yes

	option = { 
		name = "Damn Polish!"

			release = SLO
			release = TRA

			SLO = { 
				set_politics = {
					ruling_party = fascism
				}	
			}

			TRA = { 
				set_politics = {
					ruling_party = left
				}	
			}

			transfer_units_fraction = {
				target = SLO
				size = 0.32
				navy_ratio = 0
				keep_unit_leaders_trigger = { always = yes }
			}

			transfer_units_fraction = {
				target = TRA
				size = 0.32
				navy_ratio = 0
				keep_unit_leaders_trigger = { always = yes }
			}

			SOV = { country_event = { id = poland.2 days = 2 } }
	}
}

country_event = {
	id = poland.2
	title = "War begin"
	desc = "A second civil war comme in Hungary."
	picture = GFX_report_event_greater_hungary
	
	is_triggered_only = yes

	immediate = {
		every_owned_state = {
			add_core_of = HUN
			add_claim_by = HUN
			teleport_armies = {
				to_state_array = owned_controlled_states
			}
		}

		SLO = {
			transfer_technology = yes
		}
	}

	option = {
		name = "We shall fight them!"
		
	        HUN = { declare_war_on = {
		                target = SLO
		                type = annex_everything
	        } 
	        declare_war_on = {
		                target = TRA
		                type = annex_everything
	        } }

		SLO = {
			every_owned_state = {
				add_claim_by = HUN
			}
		}
		TRA = {
			every_owned_state = {
				add_claim_by = HUN
			}
		}
	}
}
